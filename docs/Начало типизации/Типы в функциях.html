Объекты в функциях

type Props = {
    name: string;
    age: number
};

function SomeFunction({ name, age }: Props) {
    return `Your are ${age} age, wow! Congratulations, ${name}`;
};

const data: Props = {
    name: "FOCKUSTY",
    age: 15
};

console.log(SomeFunction({ name: "FOCKUSTY", age: 15 }));
console.log(SomeFunction(data));
